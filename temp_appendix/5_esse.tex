Consider the metric in the form \eqref{eq:general_perturbed_FLRW_metric}, possibly in Newtonian gauge \eqref{eq:newtonian_gauge_metric}.
The action for the scalar field $\varphi$ with potential $V(\varphi)$ is
\begin{equation}
    S_\varphi = \int d^4x \sqrt{-g} \left[-\tfrac{1}{2} g^{\mu\nu}\partial_\mu\varphi \partial_\nu\varphi - V(\varphi)\right]\,, \quad X:= -g^{\mu\nu}\partial_\mu\varphi \partial_\nu\varphi\,.
\end{equation}
The energy-momentum tensor and related fluid quantities are\footnote{Note the minus sign $u_\mu = -\partial_\mu\varphi/\sqrt{X}$ so that $au^0=1+..$ matrching the fluid EFT convention.}
{\small
\begin{align}
\begin{aligned}
    &T_{\mu\nu}=-\tfrac{2}{\sqrt{-g}}\tfrac{\partial \left(\sqrt{-g}\mathcal{L}\right)}{\partial g^{\mu\nu}}
     = \partial_\mu\varphi \partial_\nu\varphi + g_{\mu\nu}\left[-\tfrac{1}{2}g^{\alpha\beta}\partial_\alpha\varphi \partial_\beta\varphi - V(\varphi)\right]
    = u_\mu u_\nu X + g_{\mu\nu}\left[\tfrac{1}{2}X + V(\varphi)\right],\\[3pt]
    &u_\mu =: -\frac{\partial_\mu\varphi}{\sqrt{X}}, \quad \rho= \tfrac12 X + V(\varphi), \quad p=\tfrac1{2}X - V(\varphi), \quad w=\frac{\tfrac1{2}X - V(\varphi)}{\tfrac1{2}X + V(\varphi)}.
\end{aligned}
\end{align}}
We expand around a homogeneous background as $\varphi(\tau,\vec{x})=\bar{\varphi}(\tau)+\delta\varphi(\tau,\vec{x})$.
Dropping mixed and second order in the metric perturbations we have 
\begin{align}\label{eq:expansion_X_V}
\begin{aligned}
    X&= -g^{\mu\nu}\partial_\mu\varphi \partial_\nu\varphi
    =\frac{1}{a^2}\bigg[(1+h^{00})\dot{\bar{\varphi}}^2 + 2\dot{\bar{\varphi}}\, \dot{\delta\varphi} + 2\dot{\bar{\varphi}}h^{0\mu}\partial_\mu\delta\varphi + (\dot{\delta\varphi})^2 - (\nabla\delta\varphi)^2\bigg]\\
    &= \frac{1}{a^2}\dot{\bar{\varphi}}^2\bigg[1+h^{00} + \frac{2\dot{\delta\varphi}}{\dot{\bar{\varphi}}} + \frac{2h^{0\mu}\partial_\mu\delta\varphi}{\dot{\bar{\varphi}}} + \frac{(\dot{\delta\varphi})^2 - (\nabla\delta\varphi)^2}{\dot{\bar{\varphi}}^2}\bigg]+O(3,h^2)\\[3pt]
    X^{-1/2}&= \frac{a}{\dot{\bar{\varphi}}}\bigg[1-\tfrac12 h^{00}-\frac{\dot{\delta\varphi}}{\dot{\bar{\varphi}}}-\frac{h^{0\mu}\partial_\mu\delta\varphi}{\dot{\bar{\varphi}}}+ \frac{(\dot{\delta\varphi})^2 +\tfrac12 (\nabla\delta\varphi)^2}{\dot{\bar{\varphi}}^2}\bigg]+O(3,h^2)\\
    V(\varphi)&= V(\bar{\varphi}) + V_\varphi(\bar{\varphi})\delta\varphi + \tfrac12 V_{\varphi\varphi}(\bar{\varphi})(\delta\varphi)^2+O(3).
\end{aligned}
\end{align}

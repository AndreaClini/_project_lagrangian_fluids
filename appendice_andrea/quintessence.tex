The action for the scalar field $\xi$ with potential $V(\xi)$ is
\begin{equation}
    S_\xi = \int d^4x \sqrt{-g} \left[-\tfrac{1}{2} g^{\mu\nu}\partial_\mu\xi \partial_\nu\xi - V(\xi)\right]\,, \quad X:= -g^{\mu\nu}\partial_\mu\xi \partial_\nu\xi\,.
\end{equation}
The energy-momentum tensor and related fluid quantities are\footnote{Note the minus sign $u_\mu = -\partial_\mu\xi/\sqrt{X}$ so that $au^0=1+..$ matrching the fluid EFT convention.}
{\small
\begin{align}
\begin{aligned}
    &T_{\mu\nu}=-\tfrac{2}{\sqrt{-g}}\tfrac{\partial \left(\sqrt{-g}\mathcal{L}\right)}{\partial g^{\mu\nu}}
     = \partial_\mu\xi \partial_\nu\xi + g_{\mu\nu}\left[-\tfrac{1}{2}g^{\alpha\beta}\partial_\alpha\xi \partial_\beta\xi - V(\xi)\right]
    = u_\mu u_\nu X + g_{\mu\nu}\left[\tfrac{1}{2}X + V(\xi)\right],\\[3pt]
    &u_\mu =: -\frac{\partial_\mu\xi}{\sqrt{X}}, \quad \rho= \tfrac12 X + V(\xi), \quad p=\tfrac1{2}X - V(\xi), \quad w=\frac{\tfrac1{2}X - V(\xi)}{\tfrac1{2}X + V(\xi)}.
\end{aligned}
\end{align}}
We expand around a homogeneous background as $\xi(\tau,\vec{x})=\bar{\xi}(\tau)+\delta\xi(\tau,\vec{x})$.
Dropping mixed and second order in the metric perturbations we have 
\begin{align}\label{eq:expansion_X_V}
\begin{aligned}
    X&= -g^{\mu\nu}\partial_\mu\xi \partial_\nu\xi
    =\frac{1}{a^2}\bigg[(1+h^{00})\dot{\bar{\xi}}^2 + 2\dot{\bar{\xi}}\, \dot{\delta\xi} + 2\dot{\bar{\xi}}h^{0\mu}\partial_\mu\delta\xi + (\dot{\delta\xi})^2 - (\nabla\delta\xi)^2\bigg]\\
    &= \frac{1}{a^2}\big(\dot{\bar{\xi}}\big)^2\bigg[1+h^{00} + \frac{2\dot{\delta\xi}}{\dot{\bar{\xi}}} + \frac{2h^{0\mu}\partial_\mu\delta\xi}{\dot{\bar{\xi}}} + \frac{(\dot{\delta\xi})^2 - (\nabla\delta\xi)^2}{\dot{\bar{\xi}}^2}\bigg]+O(3,h^2)\\[3pt]
    V(\xi)&= V(\bar{\xi}) + V_\xi(\bar{\xi})\delta\xi + \tfrac12 V_{\xi\xi}(\bar{\xi})(\delta\xi)^2+O(3).
\end{aligned}
\end{align}
Using the above expansions and \eqref{eq:general_perturbed_FLRW_metric} the quadratic action for the perturbations is then
{\small\begin{align}
\begin{aligned}
    S_\xi&= \int d^4x \sqrt{-g} \left[\tfrac12 X - V(\xi)\right]\\
    &= \int \!\!\!d^4\!x\, a^4 \!\big[1+\tfrac12 (h^{ii}-h^{00}) \big] \bigg[
        \frac{1}{2a^2}\big(\dot{\bar{\xi}}\big)^2\Big(1+h^{00} + \tfrac{2\dot{\delta\xi}}{\dot{\bar{\xi}}} + \tfrac{2h^{0\mu}\partial_\mu\delta\xi}{\dot{\bar{\xi}}} + \tfrac{(\dot{\delta\xi})^2 - (\nabla\delta\xi)^2}{\dot{\bar{\xi}}^2}\Big)\\
        &\qquad\qquad\qquad\qquad\qquad\qquad\qquad\qquad-V(\bar{\xi}) - V_\xi(\bar{\xi})\delta\xi - \tfrac12 V_{\xi\xi}(\bar{\xi})(\delta\xi)^2\bigg]\\
     &\overset{(2)}{=} \int \!\!\!d^4\!x\, a^4 \! \bigg[
        \frac{1}{2a^2}\big(\dot{\bar{\xi}}\big)^2\Big((\dot{\delta\xi}/\dot{\bar{\xi}})(h^{ii}-h^{00}) + \tfrac{2h^{0\mu}\partial_\mu\delta\xi}{\dot{\bar{\xi}}} + \tfrac{(\dot{\delta\xi})^2 - (\nabla\delta\xi)^2}{\dot{\bar{\xi}}^2}\Big)
        - \tfrac12 (h^{ii}-h^{00})V_\xi(\bar{\xi})\delta\xi - \tfrac12 V_{\xi\xi}(\bar{\xi})(\delta\xi)^2\bigg]\\
    &\overset{(2)}{=} \int \!\!\!d^4\!x\, a^4 \frac{\big(\dot{\bar{\xi}}\big)^2}{2a^2}\! \bigg[(\dot{\delta\xi}/\dot{\bar{\xi}})(h^{ii}+h^{00}) + 2h^{0j}\partial_j(\delta\xi/\dot{\bar{\xi}}) + (\dot{\delta\xi}/\dot{\bar{\xi}})^2 - (\nabla\delta\xi/\dot{\bar{\xi}})^2\\
    &\qquad\qquad\qquad- a^2 V_\xi(\bar{\xi})(h^{ii}-h^{00})(\delta\xi/\dot{\bar{\xi}}^2) - a^2 V_{\xi\xi}(\bar{\xi})(\delta\xi/\dot{\bar{\xi}})^2\bigg].
\end{aligned}\end{align}}
The fluid variables are also expanded.
Using $X^{-1/2}= (a/\dot{\bar{\xi}})\big[1-\big(\tfrac12 h^{00}+(\dot{\delta\xi}/\dot{\bar{\xi}})\big)\big]$ the 4-velocity is 
\begin{align}
    au^\mu &= -a g^{\mu\nu}\partial_\nu\xi X^{-1/2}= \big(1+\tfrac12 h^{00},\, -(\partial_j\delta\xi/\dot{\bar{\xi}}) + h^{0j}\big).
\end{align}
The fluctuations of energy density and pressure are then
{\small\begin{align}
\begin{aligned}
    &\rho = \tfrac12 X + V = \tfrac{\dot{\bar{\xi}}^2}{2a^2}+ V(\bar{\xi}) + \underbrace{\tfrac{\dot{\bar{\xi}}^2}{a^2}}_{=\bar{\rho}+\bar{p}}\bigg[(\dot{\delta\xi}/\dot{\bar{\xi}})+\tfrac12 h^{00} + a^2V_\xi(\bar{\xi})(\delta\xi/\dot{\bar{\xi}}^2) \bigg],
    \\[3pt]
    &p= \tfrac12 X - V = \tfrac{\dot{\bar{\xi}}^2}{2a^2}- V(\bar{\xi}) + \tfrac{\dot{\bar{\xi}}^2}{a^2}\bigg[(\dot{\delta\xi}/\dot{\bar{\xi}})+\tfrac12 h^{00} - a^2V_\xi(\bar{\xi})(\delta\xi/\dot{\bar{\xi}}^2) \bigg].
\end{aligned}
\end{align}}
Notably, since $p=\rho-2V$ we immediately find
{\small\begin{align}
\begin{aligned}
    \delta p =\delta \rho -2V_\xi(\bar{\xi})\delta\xi.
\end{aligned}
\end{align}}

\subsection{Matching quintessence to fluid EFT {\color{red}(in progress)}}
%============================================================================

In the 5-essence case we can express kinetic energy and potential in terms of energy density and pressure, and then equate these to the EFT expressions
\begin{align}\label{eq:identiying_rho_p_vs_scalar_field}
    \rho=\tfrac12 X +V, \quad p =\tfrac12 X -V\quad \Rightarrow\quad \begin{array}{c}
        X=\rho+p = yF_y + bF_b,\\
        V(\varphi) = \tfrac12(\rho - p) \equiv -F +\tfrac12 (yF_y+ bF_b)\,.
    \end{array}
\end{align}
The function $F$ is yet unknown.
In fact, imposing further constraints from the matching, we expect to get a set of PDEs which will single out a whole family of suitable candidates $F$.

In the scalar field case we have no notion of entropy or temperature whatsoever, as expected since it should be at temperature zero.
On the other hand, we have a well-defined notion of velocity $\vec{v}_\varphi$, with potential $v_\varphi$ or divergence $\theta_\varphi=\nabla^2v_\varphi=\vec{\nabla}\cdot\vec{v}_\varphi$, derived from the perturbations of the energy-momentum tensor
\begin{align}
    T^{\mu}_{\nu}=u^\mu u_\nu X+\delta^{\mu}_{\nu}(\tfrac12 X-V), \quad X=-\partial_\mu\varphi g^{\mu\nu}\partial_\nu\varphi, \,\, u_\mu=X^{-1/2}\partial_\mu\varphi.
\end{align}
Namely expanding $\varphi=\bar{\varphi}+\delta\varphi$ we have\todotag{check conto, guessato}
\begin{align}
    (\rho+p)\,\vec{v}_\phi\approx a^{-2}\dot{\bar{\varphi}}\vec{\nabla} \delta\varphi.
\end{align}
From \eqref{eq:peculiar_velocity_fluid_vs_phonons} we then identify
\begin{align}\label{eq:identifying_scalar_field_velocity_vs_phonon}
    v_\varphi = -\dot{\pi}_L\, \quad\text{that is}\quad \theta_\varphi=-\partial_j\dot{\pi}^j=k^2\dot{\pi}_L
\end{align}

On the other hand, we have a well-defined notion of temperature $y\equiv T$ and entropy density $s=F_y$ and comoving entropy $\sigma=F_y/b$ in the fluid EFT.
Recalling the expansions \eqref{eq:expansions_b_y} we have
{\small
\begin{align}\label{eq:expansions_b_y}
\begin{aligned}
    &\delta b^{(1)} = -\tfrac12 h^{ii}-k^2\pi_L,\\[3pt]
    &\delta y^{(1)} = \tfrac12 h^{00}+ \dot{\pi}^0 \approx \tfrac12 h^{00} +{\color{red}(\text{bkg coefficients)}} k^2 \pi_L +{\color{red}\text{(other metric terms)}}
\end{aligned}
\end{align}}
we find
\begin{align}\label{eq:identifying_entropy_vs_scala_field}
    \delta \sigma = \frac{bF_{yb}-F_y}{b} \delta b + \frac{yF_{yy}}{b}\delta y \propto \, {\color{red}(\text{bkg coefficients)}}\,k^2\pi_L+\text{(metric stuff)}
\end{align}
So that we identify $\dot{\delta\sigma}\approx \theta_\varphi$.

Let's take stock.
We have an expression for $\theta_\varphi$ in terms of the fluid EFT, and it is also immediate to identify $\delta\rho_\varphi$ and $\delta p_\varphi$ with the corresponsing fluid EFt quantities $\delta\rho$ and $\delta p$.
We should now be able to match the scalar field respecting the constraints on
\begin{align}
    w=p/\rho,\quad c_s^2:=\frac{\partial p}{\partial \rho}_{|\sigma}, \Gamma:=\frac{\partial p}{\partial \sigma}_{|\rho}
\end{align}
Indeed $\Gamma$ enter the Euler equation for the scalar field, so that, regardless of not having a notion of entropy we should be able to find an explicit expression for $\Gamma$.

Alternatively, we could simply define $c_s^2$ as the effective coefficients for fluctuations in $\ddot{\delta\varphi}=k^2c_s^2\delta\varphi+...$ and then simply subtract it off
\begin{align}
    \Gamma\delta\sigma:=\delta p-c_s^2\delta\rho
\end{align}
to get an expression for $\Gamma \delta\sigma$ in terms of 5-essence variables.

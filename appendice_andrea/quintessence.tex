The action for the scalar field $\xi$ with potential $V(\xi)$ is
\begin{equation}
    S_\xi = \int d^4x \sqrt{-g} \left[-\tfrac{1}{2} g^{\mu\nu}\partial_\mu\xi \partial_\nu\xi - V(\xi)\right]\,, \quad X:= -g^{\mu\nu}\partial_\mu\xi \partial_\nu\xi\,.
\end{equation}
The energy-momentum tensor and related fluid quantities are\footnote{Note the minus sign $u_\mu = -\partial_\mu\xi/\sqrt{X}$ so that $au^0=1+..$ matrching the fluid EFT convention.}
{\small
\begin{align}
\begin{aligned}
    &T_{\mu\nu}=-\tfrac{2}{\sqrt{-g}}\tfrac{\partial \left(\sqrt{-g}\mathcal{L}\right)}{\partial g^{\mu\nu}}
     = \partial_\mu\xi \partial_\nu\xi + g_{\mu\nu}\left[-\tfrac{1}{2}g^{\alpha\beta}\partial_\alpha\xi \partial_\beta\xi - V(\xi)\right]
    = u_\mu u_\nu X + g_{\mu\nu}\left[\tfrac{1}{2}X + V(\xi)\right],\\[3pt]
    &u_\mu =: -\frac{\partial_\mu\xi}{\sqrt{X}}, \quad \rho= \tfrac12 X + V(\xi), \quad p=\tfrac1{2}X - V(\xi), \quad w=\frac{\tfrac1{2}X - V(\xi)}{\tfrac1{2}X + V(\xi)}.
\end{aligned}
\end{align}}
We expand around a homogeneous background as $\xi(\tau,\vec{x})=\bar{\xi}(\tau)+\delta\xi(\tau,\vec{x})$.
Dropping mixed and second order in the metric perturbations we have 
\begin{align}\label{eq:expansion_X_V}
\begin{aligned}
    X&= -g^{\mu\nu}\partial_\mu\xi \partial_\nu\xi
    =\frac{1}{a^2}\bigg[(1+h^{00})\dot{\bar{\xi}}^2 + 2\dot{\bar{\xi}}\, \dot{\delta\xi} + 2\dot{\bar{\xi}}h^{0\mu}\partial_\mu\delta\xi + (\dot{\delta\xi})^2 - (\nabla\delta\xi)^2\bigg]\\
    &= \frac{1}{a^2}\big(\dot{\bar{\xi}}\big)^2\bigg[1+h^{00} + \frac{2\dot{\delta\xi}}{\dot{\bar{\xi}}} + \frac{2h^{0\mu}\partial_\mu\delta\xi}{\dot{\bar{\xi}}} + \frac{(\dot{\delta\xi})^2 - (\nabla\delta\xi)^2}{\dot{\bar{\xi}}^2}\bigg]+O(3,h^2)\\[3pt]
    V(\xi)&= V(\bar{\xi}) + V_\xi(\bar{\xi})\delta\xi + \tfrac12 V_{\xi\xi}(\bar{\xi})(\delta\xi)^2+O(3).
\end{aligned}
\end{align}
Using the above expansions and \eqref{eq:general_perturbed_FLRW_metric} the quadratic action for the perturbations is then
{\small\begin{align}
\begin{aligned}
    S_\xi&= \int d^4x \sqrt{-g} \left[\tfrac12 X - V(\xi)\right]\\
    &= \int \!\!\!d^4\!x\, a^4 \!\big[1+\tfrac12 (h^{ii}-h^{00}) \big] \bigg[
        \frac{1}{2a^2}\big(\dot{\bar{\xi}}\big)^2\Big(1+h^{00} + \tfrac{2\dot{\delta\xi}}{\dot{\bar{\xi}}} + \tfrac{2h^{0\mu}\partial_\mu\delta\xi}{\dot{\bar{\xi}}} + \tfrac{(\dot{\delta\xi})^2 - (\nabla\delta\xi)^2}{\dot{\bar{\xi}}^2}\Big)\\
        &\qquad\qquad\qquad\qquad\qquad\qquad\qquad\qquad-V(\bar{\xi}) - V_\xi(\bar{\xi})\delta\xi - \tfrac12 V_{\xi\xi}(\bar{\xi})(\delta\xi)^2\bigg]\\
     &\overset{(2)}{=} \int \!\!\!d^4\!x\, a^4 \! \bigg[
        \frac{1}{2a^2}\big(\dot{\bar{\xi}}\big)^2\Big((\dot{\delta\xi}/\dot{\bar{\xi}})(h^{ii}-h^{00}) + \tfrac{2h^{0\mu}\partial_\mu\delta\xi}{\dot{\bar{\xi}}} + \tfrac{(\dot{\delta\xi})^2 - (\nabla\delta\xi)^2}{\dot{\bar{\xi}}^2}\Big)
        - \tfrac12 (h^{ii}-h^{00})V_\xi(\bar{\xi})\delta\xi - \tfrac12 V_{\xi\xi}(\bar{\xi})(\delta\xi)^2\bigg]\\
    &\overset{(2)}{=} \int \!\!\!d^4\!x\, a^4 \frac{\big(\dot{\bar{\xi}}\big)^2}{2a^2}\! \bigg[(\dot{\delta\xi}/\dot{\bar{\xi}})(h^{ii}+h^{00}) + 2h^{0j}\partial_j(\delta\xi/\dot{\bar{\xi}}) + (\dot{\delta\xi}/\dot{\bar{\xi}})^2 - (\nabla\delta\xi/\dot{\bar{\xi}})^2\\
    &\qquad\qquad\qquad- a^2 V_\xi(\bar{\xi})(h^{ii}-h^{00})(\delta\xi/\dot{\bar{\xi}}^2) - a^2 V_{\xi\xi}(\bar{\xi})(\delta\xi/\dot{\bar{\xi}})^2\bigg].
\end{aligned}\end{align}}
The fluid variables are also expanded.
Using $X^{-1/2}= (a/\dot{\bar{\xi}})\big[1-\big(\tfrac12 h^{00}+(\dot{\delta\xi}/\dot{\bar{\xi}})\big)\big]$ and the general formula \eqref{eq:velocity_decomposition}, the 4-velocity and the peculiar velocity is 
\begin{align}\label{eq:velocity_theta_quintessence}
    au^\mu &= -a g^{\mu\nu}\partial_\nu\xi X^{-1/2}= \big(1+\tfrac12 h^{00},\, -(\partial_j\delta\xi/\dot{\bar{\xi}}) + h^{0j}\big) \quad \Rightarrow\quad
    \begin{array}{c} 
    v_j=-\partial_j\delta\xi/\dot{\bar{\xi}}\,,\\[2.5pt]
    \theta=k^2\delta\xi/\dot{\bar{\xi}}\,.
    \end{array} 
\end{align}
The fluctuations of energy density and pressure are then
{\small\begin{align}
\begin{aligned}
    &\rho = \tfrac12 X + V = \underbrace{\tfrac{\dot{\bar{\xi}}^2}{2a^2}+ V(\bar{\xi})}_{=\bar{\rho}} + \underbrace{\tfrac{\dot{\bar{\xi}}^2}{a^2}}_{=\bar{\rho}+\bar{p}}\bigg[(\dot{\delta\xi}/\dot{\bar{\xi}})+\tfrac12 h^{00} + a^2V_\xi(\bar{\xi})(\delta\xi/\dot{\bar{\xi}}^2) \bigg],
    \\[3pt]
    &p= \tfrac12 X - V = \underbrace{\tfrac{\dot{\bar{\xi}}^2}{2a^2}- V(\bar{\xi})}_{=\bar{p}} + \tfrac{\dot{\bar{\xi}}^2}{a^2}\bigg[(\dot{\delta\xi}/\dot{\bar{\xi}})+\tfrac12 h^{00} - a^2V_\xi(\bar{\xi})(\delta\xi/\dot{\bar{\xi}}^2) \bigg].
\end{aligned}
\end{align}}
Notably, since $p=\rho-2V$ we immediately find, suing \eqref{eq:velocity_theta_quintessence} for the second equality,
{\small\begin{align}\label{eq:pressure_perturbations_quintessence}
\begin{aligned}
    \delta p =\delta \rho -2V_\xi(\bar{\xi})\delta\xi = \delta \rho -2 V_\xi (k^2/\dot{\bar{\xi}})\,\theta
\end{aligned}\end{align}}

\subsection{Matching quintessence to fluid EFT {\color{red}(in progress)}}
%============================================================================

In the 5-essence case we can express kinetic energy and potential in terms of energy density and pressure, and then equate these to the EFT expressions
\begin{align}\label{eq:identiying_rho_p_vs_scalar_field}
    \rho=\tfrac12 X +V, \quad p =\tfrac12 X -V\quad \Rightarrow\quad \begin{array}{c}
        X=\rho+p = yF_y + bF_b,\\
        V(\varphi) = \tfrac12(\rho - p) \equiv -F +\tfrac12 (yF_y+ bF_b)\,.
    \end{array}
\end{align}
The function $F$ is yet unknown.
Possibly imposing further constraints , we expect to get a set of PDEs which will single out a whole family of suitable candidates $F$.
Indeed, we should also equate the expressions for fluid perturbations in the quintessence case to those in the fluid EFT.
In particular, we should identify the corresponding coefficients 
\begin{align}
    w=p/\rho,\qquad c_s^2:=\frac{\partial p}{\partial \rho}_{|\sigma},\qquad \Gamma:=\frac{\partial p}{\partial \sigma}_{|\rho}.
\end{align}
Beware in the scalar field case we have no notion of entropy or temperature whatsoever, as expected since it should be at temperature zero, and the whole fluid identification is somewhat accidental.
However, the expressions $c_a^2$ and $\Gamma$ are still defined and enter e.g. the Euler equation and the wave equation for scalar field perturbations.
Hence we should be able to relate them to the EFT formulas and get further constraints on the form of $F$.
The imposition of such constraints on $F$ is carried out in Section \ref{sec:fluids_arbitrary_eos}.

We now equate fluid perturbations in the quintessence case to the EFT expressions \eqref{eq:peculiar_velocity_fluid_vs_phonons}-\eqref{eq:rho_p_sigma_fluctuations_vs_phonon_entropy_field},
{\small\begin{subequations}\label{eq:perturbations_5essence_vs_eft}
\begin{align}
    v_j & \overset{\text{5-ess}}{=} -\partial_j\delta\xi/\dot{\bar{\xi}} + h^{0j} \overset{\text{eft}}{=} -\dot{\pi}^j + h^{0j}\,,\\[2.5pt]
    \delta\rho &\overset{\text{5-ess}}{=}(\bar{\rho}+\bar{p})\Big[(\dot{\delta\xi}/\dot{\bar{\xi}})+\tfrac12 h^{00} + a^2V_\xi(\bar{\xi})(\delta\xi/\dot{\bar{\xi}}^2) \Big]\overset{\text{eft}}{=} (\bar{\rho}+\bar{p})(\partial_j\pi^j-\tfrac12h^{ii})\,+by\,\delta\sigma\,,\\[2.5pt]
    \delta p &\overset{\text{5-ess}}{=} \delta \rho -2V_\xi(\bar{\xi})\delta\xi = \delta \rho -2 V_\xi (k^2/\dot{\bar{\xi}})\,\theta
    \overset{\text{eft}}{=}\underbrace{c_s^2}_{=1} \delta \rho + \Gamma \delta\sigma\,, \qquad \delta\sigma \overset{\text{eft}}{=} \tfrac{\dot{\bar{\varphi}}^0}{a^4by}E(\mathbf{x})\,.
\end{align}\end{subequations}}
